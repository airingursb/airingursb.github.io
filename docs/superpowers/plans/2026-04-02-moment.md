- [ ] **Step 1: Create the SQL schema file**

```sql
-- Moment content (synced from Telegram)
CREATE TABLE moments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  telegram_post_id TEXT UNIQUE NOT NULL,
  content TEXT NOT NULL,
  images JSONB DEFAULT '[]'::jsonb,
  published_at TIMESTAMPTZ NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX idx_moments_published ON moments (published_at DESC);

-- Likes (IP-based, like post_likes)
CREATE TABLE moment_likes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  moment_id UUID NOT NULL REFERENCES moments(id) ON DELETE CASCADE,
  ip TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(moment_id, ip)
);

-- Emoji reactions
CREATE TABLE moment_reactions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  moment_id UUID NOT NULL REFERENCES moments(id) ON DELETE CASCADE,
  emoji TEXT NOT NULL,
  ip TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(moment_id, emoji, ip)
);

-- Comments (mirrors post_comments pattern)
CREATE TABLE moment_comments (
  id BIGSERIAL PRIMARY KEY,
  moment_id UUID NOT NULL REFERENCES moments(id) ON DELETE CASCADE,
  parent_id BIGINT REFERENCES moment_comments(id) ON DELETE CASCADE,
  nickname TEXT NOT NULL,
  email TEXT NOT NULL,
  content TEXT NOT NULL,
  website TEXT,
  ip TEXT,
  status TEXT NOT NULL DEFAULT 'approved' CHECK (status IN ('pending', 'approved', 'spam')),
  notify BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX idx_moment_comments_moment ON moment_comments (moment_id, created_at);
CREATE INDEX idx_moment_comments_parent ON moment_comments (parent_id);
```

Note: changed `visitor_id` to `ip` to match existing blog-api convention (post_likes, comment_likes all use IP-based identification). Changed `moment_comments.id` to `BIGSERIAL` to match `post_comments` pattern.

This module follows the exact same pattern as `lib/comments.js` — lazy Supabase client via `config.blogSupabase`, exported async functions.

```javascript
import { createClient } from '@supabase/supabase-js';
import config from './config.js';

/** @type {import('@supabase/supabase-js').SupabaseClient | null} */
let _client = null;

function getClient() {
  if (_client) return _client;
  const { url, serviceKey } = config.blogSupabase;
  if (!url || !serviceKey) {
    console.warn('[moments] Missing Blog Supabase config — moment API disabled.');
    return null;
  }
  _client = createClient(url, serviceKey, { auth: { persistSession: false } });
  return _client;
}

// ── List Moments (paginated) ────────────────────────────────────────────────

export async function getMoments(page = 1, limit = 20) {
  const client = getClient();
  if (!client) return { moments: [], total: 0 };

  const offset = (page - 1) * limit;

  // Fetch moments
  const { data, error, count } = await client
    .from('moments')
    .select('id, telegram_post_id, content, images, published_at', { count: 'exact' })
    .order('published_at', { ascending: false })
    .range(offset, offset + limit - 1);

  if (error) {
    console.error('[moments] getMoments error:', error.message);
    return { moments: [], total: 0 };
  }

  if (!data || data.length === 0) return { moments: [], total: count || 0 };

  const ids = data.map((m) => m.id);

  // Batch fetch like counts
  const { data: likes } = await client
    .from('moment_likes')
    .select('moment_id')
    .in('moment_id', ids);

  const likeCounts = {};
  if (likes) {
    for (const l of likes) {
      likeCounts[l.moment_id] = (likeCounts[l.moment_id] || 0) + 1;
    }
  }

  // Batch fetch comment counts (approved only)
  const { data: comments } = await client
    .from('moment_comments')
    .select('moment_id')
    .in('moment_id', ids)
    .eq('status', 'approved');

  const commentCounts = {};
  if (comments) {
    for (const c of comments) {
      commentCounts[c.moment_id] = (commentCounts[c.moment_id] || 0) + 1;
    }
  }

  // Batch fetch reactions grouped by moment_id + emoji
  const { data: reactions } = await client
    .from('moment_reactions')
    .select('moment_id, emoji')
    .in('moment_id', ids);

  const reactionMap = {};
  if (reactions) {
    for (const r of reactions) {
      if (!reactionMap[r.moment_id]) reactionMap[r.moment_id] = {};
      reactionMap[r.moment_id][r.emoji] = (reactionMap[r.moment_id][r.emoji] || 0) + 1;
    }
  }

  const moments = data.map((m) => ({
    id: m.id,
    telegram_post_id: m.telegram_post_id,
    content: m.content,
    images: m.images || [],
    published_at: m.published_at,
    likes: likeCounts[m.id] || 0,
    comments: commentCounts[m.id] || 0,
    reactions: reactionMap[m.id] || {},
  }));

  return { moments, total: count || 0 };
}

// ── Like / Unlike ───────────────────────────────────────────────────────────

export async function toggleMomentLike(momentId, ip) {
  const client = getClient();
  if (!client) return { liked: false, count: 0 };

  // Check existing
  const { data: existing } = await client
    .from('moment_likes')
    .select('id')
    .eq('moment_id', momentId)
    .eq('ip', ip)
    .maybeSingle();

  if (existing) {
    await client.from('moment_likes').delete().eq('id', existing.id);
  } else {
    await client.from('moment_likes').insert({ moment_id: momentId, ip });
  }

  const { count } = await client
    .from('moment_likes')
    .select('id', { count: 'exact', head: true })
    .eq('moment_id', momentId);

  return { liked: !existing, count: count || 0 };
}

// ── Emoji Reactions ─────────────────────────────────────────────────────────

export async function toggleMomentReaction(momentId, emoji, ip) {
  const client = getClient();
  if (!client) return { reacted: false, reactions: {} };

  const { data: existing } = await client
    .from('moment_reactions')
    .select('id')
    .eq('moment_id', momentId)
    .eq('emoji', emoji)
    .eq('ip', ip)
    .maybeSingle();

  if (existing) {
    await client.from('moment_reactions').delete().eq('id', existing.id);
  } else {
    await client.from('moment_reactions').insert({ moment_id: momentId, emoji, ip });
  }

  // Return full reaction map for this moment
  const { data: allReactions } = await client
    .from('moment_reactions')
    .select('emoji')
    .eq('moment_id', momentId);

  const reactions = {};
  if (allReactions) {
    for (const r of allReactions) {
      reactions[r.emoji] = (reactions[r.emoji] || 0) + 1;
    }
  }

  return { reacted: !existing, reactions };
}

// ── Comments ────────────────────────────────────────────────────────────────

export async function getMomentComments(momentId) {
  const client = getClient();
  if (!client) return [];

  const { data, error } = await client
    .from('moment_comments')
    .select('id, moment_id, parent_id, nickname, email, website, content, created_at')
    .eq('moment_id', momentId)
    .eq('status', 'approved')
    .order('created_at', { ascending: true });

  if (error) {
    console.error('[moments] getMomentComments error:', error.message);
    return [];
  }

  return (data || []).map((c) => ({
    id: c.id,
    parent_id: c.parent_id,
    nickname: c.nickname,
    email: c.email,
    website: c.website || null,
    content: c.content,
    created_at: c.created_at,
  }));
}

export async function createMomentComment(momentId, nickname, email, content, parentId, ip, website, notify) {
  const client = getClient();
  if (!client) return null;

  const row = {
    moment_id: momentId,
    nickname,
    email,
    content,
    ip,
    status: 'approved',
    notify: notify || false,
  };
  if (parentId) row.parent_id = parentId;
  if (website) row.website = website;

  const { data, error } = await client
    .from('moment_comments')
    .insert(row)
    .select()
    .single();

  if (error) {
    console.error('[moments] createMomentComment error:', error.message);
    return null;
  }

  return data;
}
```

```javascript
async function handleGetMoments(req, res) {
  const url = new URL(req.url ?? '/', `http://${req.headers.host}`);
  const page = parseInt(url.searchParams.get('page') ?? '1', 10);
  const limit = Math.min(parseInt(url.searchParams.get('limit') ?? '20', 10), 50);
  const result = await getMoments(page, limit);
  json(res, 200, result);
}

async function handleToggleMomentLike(req, res, momentId) {
  const ip = getIp(req);
  const rl = checkCommentRateLimit(ip);
  if (!rl.allowed) return json(res, rl.status, { code: 'RATE_LIMITED', message: rl.message });
  const result = await toggleMomentLike(momentId, ip);
  json(res, 200, result);
}

async function handleToggleMomentReaction(req, res, momentId) {
  const ip = getIp(req);
  const rl = checkCommentRateLimit(ip);
  if (!rl.allowed) return json(res, rl.status, { code: 'RATE_LIMITED', message: rl.message });

  let body;
  try {
    body = JSON.parse(await readBody(req));
  } catch {
    return json(res, 400, { code: 'INVALID_JSON', message: 'Invalid JSON body' });
  }

  const { emoji } = body;
  if (!emoji) return json(res, 400, { code: 'MISSING_FIELDS', message: 'emoji is required' });

  const result = await toggleMomentReaction(momentId, emoji, ip);
  json(res, 200, result);
}

async function handleGetMomentComments(req, res, momentId) {
  const comments = await getMomentComments(momentId);
  json(res, 200, { comments });
}

async function handleCreateMomentComment(req, res, momentId) {
  const ip = getIp(req);
  const rl = checkCommentRateLimit(ip);
  if (!rl.allowed) return json(res, rl.status, { code: 'RATE_LIMITED', message: rl.message });

  let body;
  try {
    body = JSON.parse(await readBody(req));
  } catch {
    return json(res, 400, { code: 'INVALID_JSON', message: 'Invalid JSON body' });
  }

  const { nickname, email, content, parent_id, website, notify } = body;
  if (!nickname || !email || !content) {
    return json(res, 400, { code: 'MISSING_FIELDS', message: 'nickname, email, content are required' });
  }

  const comment = await createMomentComment(momentId, nickname, email, content, parent_id, ip, website, notify);
  if (!comment) return json(res, 500, { code: 'CREATE_FAILED', message: 'Failed to create comment' });

  json(res, 201, { comment });
}
```

This script extends the existing `parse_telegram` logic from `update_feed.py` to extract full content + images, then upserts to Supabase.

```python
#!/usr/bin/env python3
"""Sync Telegram channel messages to Supabase moments table."""

import urllib.request
import re
import os
import json
import sys
from datetime import datetime

# Load .env from repo root
_env_file = os.path.join(os.path.dirname(__file__), '..', '.env')
if os.path.exists(_env_file):
    with open(_env_file) as _f:
        for _line in _f:
            _line = _line.strip()
            if _line and not _line.startswith('#') and '=' in _line:
                _k, _v = _line.split('=', 1)
                os.environ.setdefault(_k.strip(), _v.strip())

TELEGRAM_URL = 'https://t.me/s/airingchannel'
SUPABASE_URL = os.environ.get('BLOG_SUPABASE_URL', '')
SUPABASE_KEY = os.environ.get('BLOG_SUPABASE_SERVICE_KEY', '')


def fetch_url(url):
    req = urllib.request.Request(url, headers={'User-Agent': 'Mozilla/5.0'})
    with urllib.request.urlopen(req, timeout=15) as resp:
        return resp.read().decode('utf-8', errors='replace')


def strip_html_tags(s):
    """Remove HTML tags, decode entities, normalize whitespace."""
    s = re.sub(r'<br\s*/?>', '\n', s)
    s = re.sub(r'<[^>]+>', '', s)
    s = s.replace('&amp;', '&').replace('&lt;', '<').replace('&gt;', '>')
    s = s.replace('&quot;', '"').replace('&#39;', "'").replace('&nbsp;', ' ')
    return s.strip()


def parse_telegram_full(html_str):
    """Parse Telegram public channel page for full messages with images."""
    items = []
    seen = set()

    # Split by message widget blocks
    blocks = re.split(r'(?=data-post="airingchannel/)', html_str)

    for block in blocks:
        # Extract post ID
        post_match = re.search(r'data-post="(airingchannel/\d+)"', block)
        if not post_match:
            continue
        post_id = post_match.group(1)
        if post_id in seen:
            continue
        seen.add(post_id)

        # Extract text content
        text_match = re.search(
            r'tgme_widget_message_text[^"]*"[^>]*>(.*?)</div>',
            block, re.DOTALL
        )
        content = strip_html_tags(text_match.group(1)) if text_match else ''

        # Extract images from photo wraps
        images = []
        photo_matches = re.findall(
            r"background-image:url\('([^']+)'\)",
            block
        )
        for url in photo_matches:
            if 'tgme_widget_message_photo' in block[:block.find(url) + len(url)]:
                images.append(url)

        # Skip messages with no text and no images
        if not content and not images:
            continue

        # Extract datetime
        dt_match = re.search(r'<time\s+datetime="([^"]*)"', block)
        if not dt_match:
            continue
        dt_str = dt_match.group(1)

        items.append({
            'telegram_post_id': post_id,
            'content': content,
            'images': images,
            'published_at': dt_str,
        })

    return items


def supabase_upsert(items):
    """Upsert moments to Supabase via REST API."""
    if not SUPABASE_URL or not SUPABASE_KEY:
        print('Missing SUPABASE config, skipping upsert', file=sys.stderr)
        return 0

    headers = {
        'apikey': SUPABASE_KEY,
        'Authorization': f'Bearer {SUPABASE_KEY}',
        'Content-Type': 'application/json',
        'Prefer': 'resolution=merge-duplicates',
    }

    url = f'{SUPABASE_URL}/rest/v1/moments'
    rows = []
    for item in items:
        rows.append({
            'telegram_post_id': item['telegram_post_id'],
            'content': item['content'],
            'images': json.dumps(item['images']),
            'published_at': item['published_at'],
        })

    data = json.dumps(rows).encode('utf-8')
    req = urllib.request.Request(url, data=data, headers=headers, method='POST')
    try:
        with urllib.request.urlopen(req, timeout=15) as resp:
            resp.read()
        return len(rows)
    except urllib.error.HTTPError as e:
        body = e.read().decode('utf-8', errors='replace')
        print(f'Supabase upsert error {e.code}: {body}', file=sys.stderr)
        return 0


def main():
    print('Fetching Telegram channel...')
    html = fetch_url(TELEGRAM_URL)
    items = parse_telegram_full(html)
    print(f'Parsed {len(items)} messages')

    if not items:
        print('No messages found', file=sys.stderr)
        return

    count = supabase_upsert(items)
    print(f'Upserted {count} moments to Supabase')


if __name__ == '__main__':
    main()
```

```html
<script is:inline>
(function() {
  const API = 'https://ursb.me';
  const EMOJIS = ['👍', '❤️', '🔥', '😄', '🎉', '🙏', '😢', '🤔'];
  const AVATAR = 'https://avatars.githubusercontent.com/u/10513408?v=4&s=64';

  let page = 1;
  let loading = false;
  let hasMore = true;
  let expandedId = null;

  const container = document.getElementById('momentItems');
  const loader = document.getElementById('timelineLoader');
  const endMsg = document.getElementById('timelineEnd');

  // localStorage helpers
  function getLiked() {
    try { return JSON.parse(localStorage.getItem('moment_likes') || '[]'); } catch { return []; }
  }
  function setLiked(arr) { localStorage.setItem('moment_likes', JSON.stringify(arr)); }
  function getReacted() {
    try { return JSON.parse(localStorage.getItem('moment_reactions') || '{}'); } catch { return {}; }
  }
  function setReacted(obj) { localStorage.setItem('moment_reactions', JSON.stringify(obj)); }

  // Gravatar
  function gravatarUrl(email) {
    return 'https://www.gravatar.com/avatar/?d=robohash&s=48';
  }

  // Relative time
  function timeAgo(dateStr) {
    var diff = (Date.now() - new Date(dateStr).getTime()) / 1000;
    if (diff < 60) return 'just now';
    if (diff < 3600) return Math.floor(diff / 60) + 'm ago';
    if (diff < 86400) return Math.floor(diff / 3600) + 'h ago';
    if (diff < 2592000) return Math.floor(diff / 86400) + 'd ago';
    return new Date(dateStr).toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
  }

  // ── Render a single moment card ──
  function renderMoment(m) {
    var liked = getLiked();
    var reacted = getReacted();
    var isLiked = liked.indexOf(m.id) !== -1;
    var myReactions = reacted[m.id] || [];

    var imagesHtml = '';
    if (m.images && m.images.length > 0) {
      var cls = m.images.length === 1 ? 'single' : 'multi';
      var imgs = m.images.slice(0, 4).map(function(url) {
        return '<img class="moment-img" src="' + url + '" loading="lazy" alt="" />';
      }).join('');
      imagesHtml = '<div class="moment-images ' + cls + '">' + imgs + '</div>';
    }

    var reactionsHtml = '';
    if (m.reactions && Object.keys(m.reactions).length > 0) {
      var bubbles = Object.keys(m.reactions).map(function(emoji) {
        var isReacted = myReactions.indexOf(emoji) !== -1;
        return '<span class="reaction-bubble' + (isReacted ? ' reacted' : '') + '" data-emoji="' + emoji + '">'
          + emoji + ' ' + m.reactions[emoji] + '</span>';
      }).join('');
      reactionsHtml = '<div class="moment-reactions-inline">' + bubbles + '</div>';
    }

    var html = '<div class="moment-card" data-id="' + m.id + '">'
      + '<div class="moment-dot"></div>'
      + '<div class="moment-body">'
      + '<div class="moment-header">'
      + '<img class="moment-avatar" src="' + AVATAR + '" alt="Airing" />'
      + '<span class="moment-author">Airing</span>'
      + '<span class="moment-time">· ' + timeAgo(m.published_at) + '</span>'
      + '</div>'
      + '<div class="moment-text">' + escapeHtml(m.content) + '</div>'
      + imagesHtml
      + '<div class="moment-actions">'
      + '<button class="moment-action like-btn' + (isLiked ? ' liked' : '') + '" data-id="' + m.id + '">❤️ <span class="like-count">' + (m.likes || 0) + '</span></button>'
      + '<button class="moment-action comment-btn" data-id="' + m.id + '">💬 <span class="comment-count">' + (m.comments || 0) + '</span></button>'
      + reactionsHtml
      + '</div>'
      + '<div class="moment-expanded">'
      + '<div class="emoji-picker">'
      + EMOJIS.map(function(e) { return '<button class="emoji-pick" data-emoji="' + e + '">' + e + '</button>'; }).join('')
      + '</div>'
      + '<div class="moment-comments-section" data-id="' + m.id + '"></div>'
      + '<div class="comment-form">'
      + '<div class="comment-form-row">'
      + '<input class="comment-input" placeholder="昵称" data-field="nickname" />'
      + '<input class="comment-input" placeholder="邮箱" data-field="email" />'
      + '</div>'
      + '<div class="comment-form-row">'
      + '<input class="comment-input" placeholder="说点什么..." data-field="content" style="flex:2" />'
      + '<button class="comment-submit">发送</button>'
      + '</div>'
      + '</div>'
      + '</div>'
      + '</div>'
      + '</div>';

    return html;
  }

  function escapeHtml(str) {
    if (!str) return '';
    return str.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;').replace(/\n/g, '<br/>');
  }

  // ── Fetch and render moments ──
  function loadMoments() {
    if (loading || !hasMore) return;
    loading = true;
    loader.style.display = '';

    fetch(API + '/api/moments?page=' + page + '&limit=20')
      .then(function(r) { return r.json(); })
      .then(function(data) {
        var moments = data.moments || [];
        if (moments.length === 0 || (page - 1) * 20 + moments.length >= data.total) {
          hasMore = false;
          endMsg.style.display = '';
        }
        moments.forEach(function(m) {
          container.insertAdjacentHTML('beforeend', renderMoment(m));
        });
        page++;
        loading = false;
        loader.style.display = hasMore ? '' : 'none';
      })
      .catch(function(err) {
        console.error('[moments] load error:', err);
        loading = false;
        loader.style.display = 'none';
      });
  }

  // ── Infinite scroll ──
  var scrollTimer;
  window.addEventListener('scroll', function() {
    clearTimeout(scrollTimer);
    scrollTimer = setTimeout(function() {
      if (window.innerHeight + window.scrollY >= document.body.offsetHeight - 400) {
        loadMoments();
      }
    }, 100);
  });

  // ── Event delegation ──
  container.addEventListener('click', function(e) {
    var target = e.target;

    // Like button
    var likeBtn = target.closest('.like-btn');
    if (likeBtn) {
      e.stopPropagation();
      var id = likeBtn.dataset.id;
      fetch(API + '/api/moments/' + id + '/like', { method: 'POST' })
        .then(function(r) { return r.json(); })
        .then(function(data) {
          likeBtn.querySelector('.like-count').textContent = data.count;
          var liked = getLiked();
          if (data.liked) {
            likeBtn.classList.add('liked');
            if (liked.indexOf(id) === -1) liked.push(id);
          } else {
            likeBtn.classList.remove('liked');
            liked = liked.filter(function(x) { return x !== id; });
          }
          setLiked(liked);
        });
      return;
    }

    // Comment button — toggle expand
    var commentBtn = target.closest('.comment-btn');
    if (commentBtn) {
      e.stopPropagation();
      toggleExpand(commentBtn.dataset.id);
      return;
    }

    // Emoji picker button
    var emojiBtn = target.closest('.emoji-pick');
    if (emojiBtn) {
      e.stopPropagation();
      var card = emojiBtn.closest('.moment-card');
      var momentId = card.dataset.id;
      var emoji = emojiBtn.dataset.emoji;
      fetch(API + '/api/moments/' + momentId + '/react', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ emoji: emoji }),
      })
        .then(function(r) { return r.json(); })
        .then(function(data) {
          // Update localStorage
          var reacted = getReacted();
          if (!reacted[momentId]) reacted[momentId] = [];
          if (data.reacted) {
            if (reacted[momentId].indexOf(emoji) === -1) reacted[momentId].push(emoji);
          } else {
            reacted[momentId] = reacted[momentId].filter(function(e) { return e !== emoji; });
          }
          setReacted(reacted);
          // Re-render reaction bubbles in action bar
          updateReactionBubbles(card, data.reactions, reacted[momentId] || []);
        });
      return;
    }

    // Reaction bubble click (in action bar)
    var bubble = target.closest('.reaction-bubble');
    if (bubble) {
      e.stopPropagation();
      var card = bubble.closest('.moment-card');
      var momentId = card.dataset.id;
      var emoji = bubble.dataset.emoji;
      fetch(API + '/api/moments/' + momentId + '/react', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ emoji: emoji }),
      })
        .then(function(r) { return r.json(); })
        .then(function(data) {
          var reacted = getReacted();
          if (!reacted[momentId]) reacted[momentId] = [];
          if (data.reacted) {
            if (reacted[momentId].indexOf(emoji) === -1) reacted[momentId].push(emoji);
          } else {
            reacted[momentId] = reacted[momentId].filter(function(e) { return e !== emoji; });
          }
          setReacted(reacted);
          updateReactionBubbles(card, data.reactions, reacted[momentId] || []);
        });
      return;
    }

    // Comment submit
    var submitBtn = target.closest('.comment-submit');
    if (submitBtn) {
      e.stopPropagation();
      var form = submitBtn.closest('.comment-form');
      var card = submitBtn.closest('.moment-card');
      var momentId = card.dataset.id;
      var nickname = form.querySelector('[data-field="nickname"]').value.trim();
      var email = form.querySelector('[data-field="email"]').value.trim();
      var content = form.querySelector('[data-field="content"]').value.trim();
      if (!nickname || !email || !content) return;

      submitBtn.disabled = true;
      fetch(API + '/api/moments/' + momentId + '/comments', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ nickname: nickname, email: email, content: content }),
      })
        .then(function(r) { return r.json(); })
        .then(function(data) {
          if (data.comment) {
            form.querySelector('[data-field="content"]').value = '';
            loadComments(momentId);
            // Update comment count
            var countEl = card.querySelector('.comment-count');
            countEl.textContent = parseInt(countEl.textContent) + 1;
          }
          submitBtn.disabled = false;
        })
        .catch(function() { submitBtn.disabled = false; });
      return;
    }

    // Reply button
    var replyBtn = target.closest('.comment-reply-btn');
    if (replyBtn) {
      e.stopPropagation();
      var card = replyBtn.closest('.moment-card');
      var form = card.querySelector('.comment-form');
      var contentInput = form.querySelector('[data-field="content"]');
      var nick = replyBtn.dataset.nick;
      contentInput.value = '@' + nick + ' ';
      contentInput.focus();
      return;
    }

    // Card click — toggle expand
    var card = target.closest('.moment-card');
    if (card && !target.closest('.moment-expanded') && !target.closest('.moment-actions')) {
      toggleExpand(card.dataset.id);
    }
  });

  function toggleExpand(id) {
    var cards = container.querySelectorAll('.moment-card');
    cards.forEach(function(c) {
      if (c.dataset.id === id) {
        c.classList.toggle('expanded');
        if (c.classList.contains('expanded')) {
          loadComments(id);
        }
      } else {
        c.classList.remove('expanded');
      }
    });
  }

  function loadComments(momentId) {
    var section = container.querySelector('.moment-comments-section[data-id="' + momentId + '"]');
    if (!section) return;
    section.innerHTML = '<div class="comments-title">Loading...</div>';

    fetch(API + '/api/moments/' + momentId + '/comments')
      .then(function(r) { return r.json(); })
      .then(function(data) {
        var comments = data.comments || [];
        if (comments.length === 0) {
          section.innerHTML = '<div class="comments-title">暂无评论</div>';
          return;
        }

        // Build tree
        var roots = [];
        var childMap = {};
        comments.forEach(function(c) {
          if (c.parent_id) {
            if (!childMap[c.parent_id]) childMap[c.parent_id] = [];
            childMap[c.parent_id].push(c);
          } else {
            roots.push(c);
          }
        });

        var html = '<div class="comments-title">评论 (' + comments.length + ')</div>';
        roots.forEach(function(c) {
          html += renderComment(c, childMap);
        });
        section.innerHTML = html;
      });
  }

  function renderComment(c, childMap) {
    var children = childMap[c.id] || [];
    var repliesHtml = '';
    if (children.length > 0) {
      repliesHtml = '<div class="comment-replies">';
      children.forEach(function(child) {
        repliesHtml += renderComment(child, childMap);
      });
      repliesHtml += '</div>';
    }

    return '<div class="comment-item">'
      + '<img class="comment-avatar" src="' + gravatarUrl(c.email) + '" alt="" />'
      + '<div>'
      + '<div class="comment-meta">'
      + '<span class="comment-nick">' + escapeHtml(c.nickname) + '</span>'
      + '<span class="comment-date">' + timeAgo(c.created_at) + '</span>'
      + '</div>'
      + '<div class="comment-text">' + escapeHtml(c.content) + '</div>'
      + '<button class="comment-reply-btn" data-nick="' + escapeHtml(c.nickname) + '">回复</button>'
      + repliesHtml
      + '</div>'
      + '</div>';
  }

  function updateReactionBubbles(card, reactions, myReactions) {
    var actionsEl = card.querySelector('.moment-actions');
    var existingBubbles = actionsEl.querySelector('.moment-reactions-inline');
    if (existingBubbles) existingBubbles.remove();

    if (reactions && Object.keys(reactions).length > 0) {
      var bubbles = Object.keys(reactions).map(function(emoji) {
        var isReacted = myReactions.indexOf(emoji) !== -1;
        return '<span class="reaction-bubble' + (isReacted ? ' reacted' : '') + '" data-emoji="' + emoji + '">'
          + emoji + ' ' + reactions[emoji] + '</span>';
      }).join('');
      actionsEl.insertAdjacentHTML('beforeend', '<div class="moment-reactions-inline">' + bubbles + '</div>');
    }
  }

  // ── Init ──
  loadMoments();
})();
</script>
```
