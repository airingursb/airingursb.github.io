# AI Chat Implementation Plan (SHU-201)

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an AI chat feature embedded in the intro card of the homepage, backed by an OpenClaw middleware service with Supabase storage.

**Architecture:** Frontend chat UI in index.html talks to a Node.js middleware service (in Docker), which proxies to OpenClaw Gateway and persists conversations in Supabase. Visitor identity uses HMAC-signed tokens. SSE streaming for real-time responses.

**Tech Stack:** Vanilla JS + SSE (frontend), Node.js with native http (middleware), Supabase PostgreSQL, OpenClaw Gateway, Docker/OrbStack, LocalCan.

- Message sending with SSE streaming
- UI state management (rounds, errors, limit)

```javascript
// ── AI Chat ──
(function () {
  const CHAT_API_BASE = ''; // TODO: set to LocalCan public domain
  const MAX_ROUNDS = 5;
  const MAX_INPUT_LENGTH = 200;
  const SSE_TIMEOUT_MS = 30000;

  const chatArea = document.getElementById('chatArea');
  const chatInput = document.getElementById('chatInput');
  const chatError = document.getElementById('chatError');
  const chatRounds = document.getElementById('chatRounds');

  let visitorToken = localStorage.getItem('chat_visitor_token');
  let currentRound = 0;
  let isStreaming = false;

  // ── Token Management ──
  async function ensureToken() {
    if (visitorToken) return visitorToken;
    try {
      const res = await fetch(`${CHAT_API_BASE}/api/chat/init`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' }
      });
      if (!res.ok) throw new Error('Init failed');
      const data = await res.json();
      visitorToken = data.visitor_token;
      localStorage.setItem('chat_visitor_token', visitorToken);
      return visitorToken;
    } catch (e) {
      showError('⚠ 连接失败，请稍后再试');
      return null;
    }
  }

  // ── History Loading ──
  async function loadHistory() {
    if (!visitorToken) return;
    try {
      const res = await fetch(`${CHAT_API_BASE}/api/chat/history`, {
        headers: { 'X-Visitor-Token': visitorToken }
      });
      if (!res.ok) {
        if (res.status === 401) {
          localStorage.removeItem('chat_visitor_token');
          visitorToken = null;
        }
        return;
      }
      const data = await res.json();
      currentRound = data.round || 0;
      updateRoundsDisplay();
      if (data.messages && data.messages.length > 0) {
        data.messages.forEach(msg => appendMessage(msg.role, msg.content, false));
        chatArea.classList.add('active');
        chatArea.scrollTop = chatArea.scrollHeight;
      }
      if (currentRound >= MAX_ROUNDS) {
        showLimitReached();
      }
    } catch (e) {
      // Silent fail on history load
    }
  }

  // ── Message Display ──
  function appendMessage(role, content, animate) {
    const div = document.createElement('div');
    div.className = 'chat-msg';

    const roleSpan = document.createElement('span');
    roleSpan.className = role === 'user' ? 'chat-role-user' : 'chat-role-ai';
    roleSpan.textContent = role === 'user' ? 'you> ' : 'ai> ';

    const contentSpan = document.createElement('span');
    contentSpan.className = 'chat-content';
    if (animate) {
      contentSpan.classList.add('chat-typing');
    } else {
      contentSpan.innerHTML = renderMarkdown(content);
    }

    div.appendChild(roleSpan);
    div.appendChild(contentSpan);
    chatArea.appendChild(div);
    chatArea.classList.add('active');
    chatArea.scrollTop = chatArea.scrollHeight;

    return contentSpan;
  }

  function renderMarkdown(text) {
    // Lightweight markdown: bold, italic, code, links
    return text
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/`([^`]+)`/g, '<code>$1</code>')
      .replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>')
      .replace(/\*([^*]+)\*/g, '<em>$1</em>')
      .replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2" target="_blank" rel="noopener">$1</a>');
  }

  // ── Send Message ──
  async function sendMessage(message) {
    if (isStreaming || !message.trim()) return;
    if (currentRound >= MAX_ROUNDS) {
      showLimitReached();
      return;
    }

    clearError();
    isStreaming = true;
    chatInput.disabled = true;
    chatInput.value = '';

    // Show user message
    appendMessage('user', message, false);

    // Show AI placeholder with typing indicator
    const aiContent = appendMessage('assistant', '', true);

    const token = await ensureToken();
    if (!token) {
      isStreaming = false;
      chatInput.disabled = false;
      aiContent.parentElement.remove();
      return;
    }

    let fullResponse = '';
    let timeoutId;

    try {
      const controller = new AbortController();
      timeoutId = setTimeout(() => controller.abort(), SSE_TIMEOUT_MS);

      const res = await fetch(`${CHAT_API_BASE}/api/chat`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ visitor_token: token, message: message.trim() }),
        signal: controller.signal
      });

      clearTimeout(timeoutId);

      if (!res.ok) {
        const errData = await res.json().catch(() => ({}));
        if (errData.code === 'RATE_LIMITED') {
          showError(`⚠ 请求太频繁，请 ${Math.ceil((errData.retry_after || 300) / 60)} 分钟后再试`);
        } else if (errData.code === 'LIMIT_REACHED') {
          currentRound = MAX_ROUNDS;
          showLimitReached();
        } else if (errData.code === 'INVALID_TOKEN') {
          localStorage.removeItem('chat_visitor_token');
          visitorToken = null;
          showError('⚠ 会话已过期，请刷新页面');
        } else {
          showError('⚠ AI 服务暂时不可用');
        }
        aiContent.parentElement.remove();
        isStreaming = false;
        chatInput.disabled = false;
        return;
      }

      const reader = res.body.getReader();
      const decoder = new TextDecoder();
      let buffer = '';

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split('\n');
        buffer = lines.pop();

        for (const line of lines) {
          if (!line.startsWith('data: ')) continue;
          const data = line.slice(6);
          if (data === '[DONE]') break;

          try {
            const parsed = JSON.parse(data);
            if (parsed.type === 'delta' && parsed.content) {
              fullResponse += parsed.content;
              aiContent.textContent = fullResponse;
              chatArea.scrollTop = chatArea.scrollHeight;
            } else if (parsed.type === 'done') {
              currentRound = parsed.usage?.round || currentRound + 1;
              updateRoundsDisplay();
            } else if (parsed.type === 'error') {
              showError(`⚠ ${parsed.message}`);
            }
          } catch (e) {
            // Skip unparseable lines
          }
        }
      }

      // Render final markdown
      aiContent.classList.remove('chat-typing');
      aiContent.innerHTML = renderMarkdown(fullResponse);

      if (currentRound >= MAX_ROUNDS) {
        showLimitReached();
      }
    } catch (e) {
      clearTimeout(timeoutId);
      if (e.name === 'AbortError') {
        showError('⚠ AI 服务响应超时');
      } else {
        showError('⚠ 连接失败，请稍后再试');
      }
      if (!fullResponse) {
        aiContent.parentElement.remove();
      } else {
        aiContent.classList.remove('chat-typing');
        aiContent.innerHTML = renderMarkdown(fullResponse);
      }
    } finally {
      isStreaming = false;
      chatInput.disabled = false;
      chatInput.focus();
    }
  }

  // ── UI Helpers ──
  function updateRoundsDisplay() {
    chatRounds.textContent = currentRound > 0 ? `${currentRound}/${MAX_ROUNDS}` : '';
  }

  function showError(msg) {
    chatError.textContent = msg;
  }

  function clearError() {
    chatError.textContent = '';
  }

  function showLimitReached() {
    chatInput.disabled = true;
    chatInput.style.display = 'none';
    document.querySelector('.chat-prompt').style.display = 'none';
    const limitMsg = document.createElement('div');
    limitMsg.className = 'chat-limit-msg';
    limitMsg.innerHTML = '今日对话次数已用完，24 小时后重置。<a href="https://blog.ursb.me" target="_blank">访问博客了解更多 →</a>';
    document.querySelector('.chat-input-row').appendChild(limitMsg);
  }

  // ── Event Binding ──
  chatInput.addEventListener('keydown', (e) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      const msg = chatInput.value.trim();
      if (msg && msg.length <= MAX_INPUT_LENGTH) {
        sendMessage(msg);
      }
    }
  });

  // ── Init ──
  loadHistory();
})();
```

```javascript
import { readFileSync } from 'fs';

function loadEnv() {
  try {
    const content = readFileSync(new URL('../.env', import.meta.url), 'utf8');
    for (const line of content.split('\n')) {
      const trimmed = line.trim();
      if (!trimmed || trimmed.startsWith('#')) continue;

      const val = trimmed.slice(idx + 1).trim();
      if (!process.env[key]) process.env[key] = val;

  } catch (e) {
    // .env file is optional
  }
}
```

```javascript
// services/chat/lib/auth.js
import { createHmac, randomUUID } from 'crypto';
import { config } from './config.js';

function sign(visitorId) {
  return createHmac('sha256', config.hmacSecret)
    .update(visitorId)
    .digest('hex');
}

export function createToken() {
  const visitorId = randomUUID();
  const signature = sign(visitorId);
  return { visitorId, token: `${visitorId}.${signature}` };
}

export function verifyToken(token) {
  if (!token || typeof token !== 'string') return null;
  const dotIdx = token.indexOf('.');
  if (dotIdx === -1) return null;
  const visitorId = token.slice(0, dotIdx);
  const sig = token.slice(dotIdx + 1);
  const expected = sign(visitorId);
  // Constant-time comparison
  if (sig.length !== expected.length) return null;
  let diff = 0;
  for (let i = 0; i < sig.length; i++) {
    diff |= sig.charCodeAt(i) ^ expected.charCodeAt(i);
  }
  return diff === 0 ? visitorId : null;
}
```

```javascript
// services/chat/lib/rate-limit.js
import { config } from './config.js';

const store = new Map();
const WINDOW_MS = 60 * 60 * 1000; // 1 hour

// Clean up expired entries every 10 minutes
setInterval(() => {
  const now = Date.now();
  for (const [ip, timestamps] of store) {
    const valid = timestamps.filter(t => now - t < WINDOW_MS);
    if (valid.length === 0) store.delete(ip);
    else store.set(ip, valid);
  }
}, 10 * 60 * 1000).unref();

export function checkRateLimit(ip) {
  const now = Date.now();
  const timestamps = (store.get(ip) || []).filter(t => now - t < WINDOW_MS);

  if (timestamps.length >= config.rateLimitPerHour) {
    const oldestValid = timestamps[0];
    const retryAfter = Math.ceil((WINDOW_MS - (now - oldestValid)) / 1000);
    return { allowed: false, retryAfter };
  }

  timestamps.push(now);
  store.set(ip, timestamps);
  return { allowed: true };
}
```

```sql
  visitor_id TEXT UNIQUE NOT NULL,

  user_agent TEXT,
  first_seen_at TIMESTAMPTZ DEFAULT NOW(),

-- Messages
CREATE TABLE IF NOT EXISTS chat_messages (
  id BIGSERIAL PRIMARY KEY,
  visitor_id TEXT NOT NULL REFERENCES chat_visitors(visitor_id),
  role TEXT NOT NULL CHECK (role IN ('user', 'assistant')),
  content TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_messages_visitor ON chat_messages(visitor_id, created_at);
```

```javascript
// services/chat/lib/supabase.js
import { createClient } from '@supabase/supabase-js';
import { config } from './config.js';

let supabase = null;

function getClient() {
  if (!supabase) {
    if (!config.supabase.url || !config.supabase.serviceKey) {
      console.warn('[supabase] Missing SUPABASE_URL or SUPABASE_SERVICE_KEY');
      return null;
    }
    supabase = createClient(config.supabase.url, config.supabase.serviceKey);
  }
  return supabase;
}

export async function createVisitor(visitorId, ip, userAgent) {
  const client = getClient();
  if (!client) return;
  const { error } = await client.from('chat_visitors').upsert({
    visitor_id: visitorId,
    ip,
    user_agent: userAgent,
    last_seen_at: new Date().toISOString(),
  }, { onConflict: 'visitor_id' });
  if (error) console.error('[supabase] createVisitor error:', error.message);
}

export async function getRecentMessages(visitorId) {
  const client = getClient();
  if (!client) return { messages: [], round: 0 };

  const since = new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString();
  const { data, error } = await client
    .from('chat_messages')
    .select('role, content, created_at')
    .eq('visitor_id', visitorId)
    .gte('created_at', since)
    .order('created_at', { ascending: true })
    .limit(10);

  if (error) {
    console.error('[supabase] getRecentMessages error:', error.message);
    return { messages: [], round: 0 };
  }

  const messages = data || [];
  const round = Math.floor(messages.filter(m => m.role === 'user').length);
  return { messages, round };
}

export async function saveMessages(visitorId, userMessage, assistantMessage) {
  const client = getClient();
  if (!client) return;
  const { error } = await client.from('chat_messages').insert([
    { visitor_id: visitorId, role: 'user', content: userMessage },
    { visitor_id: visitorId, role: 'assistant', content: assistantMessage },
  ]);
  if (error) console.error('[supabase] saveMessages error:', error.message);
}

export async function updateVisitorActivity(visitorId, ip, userAgent) {
  const client = getClient();
  if (!client) return;
  const { error } = await client
    .from('chat_visitors')
    .update({ last_seen_at: new Date().toISOString(), ip, user_agent: userAgent })
    .eq('visitor_id', visitorId);
  if (error) console.error('[supabase] updateVisitorActivity error:', error.message);
}
```

```javascript
export async function streamChat(messages, visitorId) {
  const url = `${config.openclaw.url}/v1/chat/completions`;
  const res = await fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${config.openclaw.token}`,
      'x-openclaw-agent-id': config.openclaw.agentId,
    },
    body: JSON.stringify({
      model: `openclaw:${config.openclaw.agentId}`,
      messages,
      stream: true,
      user: visitorId,
      max_tokens: 500,
    }),
  });

  if (!res.ok) {
    const text = await res.text().catch(() => '');
    throw new Error(`OpenClaw error ${res.status}: ${text}`);
  }

  return res.body;
}
```

```javascript
// services/chat/server.js
import { createServer } from 'http';
import { config } from './lib/config.js';
import { createToken, verifyToken } from './lib/auth.js';
import { checkRateLimit } from './lib/rate-limit.js';
import { createVisitor, getRecentMessages, saveMessages, updateVisitorActivity } from './lib/supabase.js';

const server = createServer(async (req, res) => {
  const ip = req.headers['x-forwarded-for']?.split(',')[0]?.trim() || req.socket.remoteAddress;
  const origin = req.headers.origin;

  // CORS
  if (origin && config.allowedOrigins.includes(origin)) {
    res.setHeader('Access-Control-Allow-Origin', origin);
    res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
    res.setHeader('Access-Control-Allow-Headers', 'Content-Type, X-Visitor-Token');
    res.setHeader('Access-Control-Max-Age', '86400');
  }

  if (req.method === 'OPTIONS') {
    res.writeHead(204);
    res.end();
    return;
  }

  const url = new URL(req.url, `http://${req.headers.host}`);

  try {
    if (url.pathname === '/api/chat/init' && req.method === 'POST') {
      await handleInit(req, res, ip);
    } else if (url.pathname === '/api/chat' && req.method === 'POST') {
      await handleChat(req, res, ip);
    } else if (url.pathname === '/api/chat/history' && req.method === 'GET') {
      await handleHistory(req, res);
    } else if (url.pathname === '/health' && req.method === 'GET') {
      json(res, 200, { status: 'ok' });
    } else {
      json(res, 404, { error: 'Not found' });
    }
  } catch (e) {
    console.error('[server] Unhandled error:', e);
    json(res, 500, { type: 'error', code: 'INTERNAL', message: 'Internal server error' });
  }
});

// ── Route Handlers ──

async function handleInit(req, res, ip) {
  const { visitorId, token } = createToken();
  await createVisitor(visitorId, ip, req.headers['user-agent']);
  json(res, 200, { visitor_token: token });
}

async function handleChat(req, res, ip) {
  // Content-Type check
  const ct = req.headers['content-type'] || '';
  if (!ct.includes('application/json')) {
    json(res, 400, { type: 'error', code: 'BAD_REQUEST', message: 'Content-Type must be application/json' });
    return;
  }

  // Parse body
  const body = await readBody(req);
  if (!body) {
    json(res, 400, { type: 'error', code: 'BAD_REQUEST', message: 'Invalid JSON' });
    return;
  }

  // Verify token
  const visitorId = verifyToken(body.visitor_token);
  if (!visitorId) {
    json(res, 401, { type: 'error', code: 'INVALID_TOKEN', message: '无效的访客凭证' });
    return;
  }

  // Rate limit
  const rateCheck = checkRateLimit(ip);
  if (!rateCheck.allowed) {
    json(res, 429, { type: 'error', code: 'RATE_LIMITED', message: '请稍后再试', retry_after: rateCheck.retryAfter });
    return;
  }

  // Validate message
  const message = (body.message || '').trim();
  if (!message || message.length > config.maxInputLength) {
    json(res, 400, { type: 'error', code: 'BAD_REQUEST', message: `消息不能为空且不超过 ${config.maxInputLength} 字符` });
    return;
  }

  // Check round limit
  const { messages: history, round } = await getRecentMessages(visitorId);
  if (round >= config.maxRounds) {
    json(res, 429, { type: 'error', code: 'LIMIT_REACHED', message: '今日对话次数已用完，24 小时后重置' });
    return;
  }

  // Build messages array for OpenClaw
  const openclawMessages = history.map(m => ({ role: m.role, content: m.content }));
  openclawMessages.push({ role: 'user', content: message });

  // Stream response
  res.writeHead(200, {
    'Content-Type': 'text/event-stream',
    'Cache-Control': 'no-cache',
    'Connection': 'keep-alive',
    ...(res.getHeader('Access-Control-Allow-Origin') ? {} : {}),
  });

  let fullResponse = '';

  try {
    const { streamChat } = await import('./lib/openclaw.js');
    const stream = await streamChat(openclawMessages, visitorId);

    const reader = stream.getReader();
    const decoder = new TextDecoder();
    let buffer = '';

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split('\n');
      buffer = lines.pop();

      for (const line of lines) {
        if (!line.startsWith('data: ')) continue;
        const data = line.slice(6);
        if (data === '[DONE]') continue;

        try {
          const parsed = JSON.parse(data);
          const delta = parsed.choices?.[0]?.delta?.content;
          if (delta) {
            fullResponse += delta;
            res.write(`data: ${JSON.stringify({ type: 'delta', content: delta })}\n\n`);
          }
        } catch (e) {
          // Skip
        }
      }
    }

    const newRound = round + 1;
    res.write(`data: ${JSON.stringify({ type: 'done', usage: { round: newRound, max_rounds: config.maxRounds } })}\n\n`);
    res.write('data: [DONE]\n\n');
    res.end();

    // Save to Supabase (fire-and-forget)
    saveMessages(visitorId, message, fullResponse).catch(e =>
      console.error('[server] Failed to save messages:', e.message)
    );
    updateVisitorActivity(visitorId, ip, req.headers['user-agent']).catch(() => {});
  } catch (e) {
    console.error('[server] OpenClaw stream error:', e);
    if (!res.headersSent) {
      json(res, 502, { type: 'error', code: 'GATEWAY_ERROR', message: 'AI 服务暂时不可用' });
    } else {
      res.write(`data: ${JSON.stringify({ type: 'error', message: 'AI 服务暂时不可用' })}\n\n`);
      res.end();
    }
  }
}

async function handleHistory(req, res) {
  const token = req.headers['x-visitor-token'];
  const visitorId = verifyToken(token);
  if (!visitorId) {
    json(res, 401, { type: 'error', code: 'INVALID_TOKEN', message: '无效的访客凭证' });
    return;
  }

  const { messages, round } = await getRecentMessages(visitorId);
  json(res, 200, { messages, round, max_rounds: config.maxRounds });
}

// ── Helpers ──

function json(res, status, data) {
  res.writeHead(status, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify(data));
}

function readBody(req) {
  return new Promise((resolve) => {
    let data = '';
    req.on('data', chunk => { data += chunk; if (data.length > 10000) resolve(null); });
    req.on('end', () => { try { resolve(JSON.parse(data)); } catch { resolve(null); } });
    req.on('error', () => resolve(null));
  });
}

// ── Start ──

server.listen(config.port, () => {
  console.log(`[chat-middleware] Listening on :${config.port}`);
  console.log(`[chat-middleware] Allowed origins: ${config.allowedOrigins.join(', ')}`);
});
```
